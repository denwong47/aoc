1use crate::{SimpleGraphError, traits, wrapper};

4    cmp::{Ord, Reverse},
5    collections::{BinaryHeap, HashMap, HashSet},
6    fmt::Debug,
7    hash::Hash,

10/// Implements Dijkstra's algorithm to find the shortest path from a start node to a destination node.
11pub fn dijkstra<'s, K, D, N>(
12    start: &'s N,
13    destination: &'s K,
14    get_node_by_key: impl Fn(&K) -> Option<&'s N> + Clone,
15) -> Result<(Vec<&'s K>, D), SimpleGraphError<K, D>>
16where
17    K: Debug + Clone + Eq + Hash + 's,
18    D: Zero + Ord + Clone + Debug,
19    N: traits::IsNode<'s, K, D>,
20{
21    let mut current_node = start;
22    let mut visited_nodes: HashSet<&'s K> = HashSet::new();
23    let mut unvisited_nodes: HashMap<&'s K, (&'s N, Vec<&'s K>, D)> = HashMap::new();
24    let mut unvisited_distances: BinaryHeap<(Reverse<D>, wrapper::UnorderedItem<&'s K>)> =
25        BinaryHeap::new();
26
27    unvisited_nodes.insert(
28        current_node.id(),
29        (current_node, vec![current_node.id()], D::zero()),
30    );
31
32    loop {
33        // Mark the current node as visited
34        visited_nodes.insert(current_node.id());
35        #[cfg(feature = "trace")]
36        eprintln!("Visiting node {:?}", current_node.id());
37        let (current_path, current_distance) = match unvisited_nodes.remove(current_node.id()) {
38            Some((_, path, distance)) => Ok((path, distance)),
39            None => Err(SimpleGraphError::Unreachable(format!(
40                "Current node {:?} not in unvisited nodes",
41                current_node.id()
42            ))),
43        }?;
44
45        #[cfg(feature = "trace")]
46        eprintln!(
47            "Visiting node {:?} with current distance {:?} and path {:?}",
48            current_node.id(),
49            current_distance,
50            current_path
51        );
52
53        // Stage 1 - Check if we reached the destination
54        if current_node.id() == destination {
55            return Ok((current_path, current_distance));
56        }
57
58        // Stage 2 - Update neighbours
59        current_node
60            .neighbours(get_node_by_key.clone())
61            .try_for_each(|(neighbour_node, distance)| {
62                let neighbour_id = neighbour_node.id();
63                if distance < D::zero() {
64                    return Err(SimpleGraphError::NegativeDistance {
65                        start: current_node.id().clone(),
66                        destination: neighbour_id.clone(),
67                        distance: distance.clone(),
68                    });
69                }
70
71                if visited_nodes.contains(neighbour_id) {
72                    #[cfg(feature = "trace")]
73                    eprintln!("Neighbour node {neighbour_id:?} already visited, skipping",);
74
75                    return Ok(());
76                }
77
78                let new_distance = current_distance.clone() + distance.clone();
79                unvisited_nodes
80                    .entry(neighbour_id)
81                    .and_modify(|(_, path, existing_distance)| {
82                        #[cfg(feature = "trace")]
83                        eprintln!(
84                            "Updating neighbour node {neighbour_id:?} with a shorter distance of {distance:?} (existing: {existing_distance:?})",
85                        );
86
87                        // Update the path and distance if the new distance is shorter
88                        if new_distance < *existing_distance {
89                            *existing_distance = new_distance.clone();
90
91                            let mut new_path = current_path.clone();
92                            new_path.push(neighbour_id);
93                            std::mem::swap(path, &mut new_path);
94                        }
95                    })
96                    .or_insert_with(|| {
97                        #[cfg(feature = "trace")]
98                        eprintln!(
99                            "Adding new neighbour node {neighbour_id:?} with distance {distance:?}",
100                        );
101                        // Create a new entry for this neighbour if it doesn't exist
102                        let mut new_path = current_path.clone();
103                        new_path.push(neighbour_id);
104                        (neighbour_node, new_path, new_distance.clone())
105                    });
106
107                // Push the new distance to the priority queue.
108                // We do not check for existing entries here; they will be ignored when popped if outdated.
109                unvisited_distances.push((
110                    Reverse(new_distance),
111                    wrapper::UnorderedItem::new(neighbour_id),
112                ));
113
114                Ok(())
115            })?;
116
117        // Stage 3 - Select the next current node
118        loop {
119            match unvisited_distances.pop() {
120                Some((_, wrapper::UnorderedItem(neighbour_id))) => {
121                    if visited_nodes.contains(neighbour_id) {
122                        #[cfg(feature = "trace")]
123                        eprintln!("Neighbour node {neighbour_id:?} already visited, skipping",);
124
125                        continue;
126                    }
127                    if let Some((neighbour_node, _, _)) = unvisited_nodes.get(neighbour_id) {
128                        current_node = *neighbour_node;
129                        #[cfg(feature = "trace")]
130                        eprintln!("Next current node set to {:?}", current_node.id());
131                        break;
132                    } else {
133                        return Err(SimpleGraphError::Unreachable(format!(
134                            "Neighbour node {:?} not found in unvisited nodes",
135                            neighbour_id
136                        )));
137                    }
138                }
139                None => {
140                    return Err(SimpleGraphError::Unreachable(format!(
141                        "Destination node {:?} is unreachable from start node {:?}",
142                        destination,
143                        start.id()
144                    )));
145                }
146            }
147        }
148
149        #[cfg(feature = "trace")]
150        eprintln!(
151            "Unvisited nodes remaining: {:?}",
152            unvisited_nodes.keys().collect::<Vec<&&K>>()
153        );
154    }
155}
