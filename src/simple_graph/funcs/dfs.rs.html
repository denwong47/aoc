7pub struct NodeInProgress<'s, K, D, N> {
8    node: &'s N,
9    distance: D,
10    next_index_to_visit: usize,
11    _phantom: std::marker::PhantomData<K>,
12}

14impl<'s, K, D, N> std::fmt::Debug for NodeInProgress<'s, K, D, N>
15where
16    D: Debug,
17    N: std::fmt::Debug,
18{
19    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
20        write!(f, "NodeInProgress {{ node: ")
21            .and_then(|_| std::fmt::Debug::fmt(self.node, f))
22            .and_then(|_| write!(f, ", distance: {:?}, ", self.distance))
23            .and_then(|_| write!(f, "next_index_to_visit: {} }}", self.next_index_to_visit))
24    }
25}

27impl<'s, K, D, N> NodeInProgress<'s, K, D, N>
28where
29    K: Debug + Clone + Eq + Hash + 's,
30    D: Zero + Ord + Clone + Debug,
31    N: traits::IsNodeWithIndexedNeighbours<'s, K, D>,
32{
33    pub fn new(node: &'s N, distance: D) -> Self {
34        Self {
35            node,
36            distance,
37            next_index_to_visit: 0,
38            _phantom: std::marker::PhantomData,
39        }
40    }
41
42    pub fn next_unvisited_neighbour(
43        &mut self,
44        get_node_by_key: impl Fn(&K) -> Option<&'s N>,
45    ) -> Option<Self> {
46        self.node
47            .get_neighbour(self.next_index_to_visit, get_node_by_key)
48            .map(|(node, distance)| {
49                // Advance index for next call
50                self.next_index_to_visit += 1;
51                Self::new(node, self.distance.clone() + distance)
52            })
53    }
54}

56pub struct Dfs<'s, K, D, N>
57where
58    K: Debug + Clone + Eq + Hash + 's,
59    D: Zero + Ord + Clone + Debug,
60    N: traits::IsNode<'s, K, D>,
61{
62    start: &'s N,
63    destination: &'s N,
64    tracker: Vec<NodeInProgress<'s, K, D, N>>,
65}
66
67impl<'s, K, D, N> std::fmt::Debug for Dfs<'s, K, D, N>
68where
69    K: Debug + Clone + Eq + Hash + 's,
70    D: Zero + Ord + Clone + Debug,
71    N: traits::IsNode<'s, K, D>,
72{
73    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
74        write!(f, "Dfs {{ start: ")
75            .and_then(|_| std::fmt::Debug::fmt(self.start.id(), f))
76            .and_then(|_| write!(f, ", destination: "))
77            .and_then(|_| std::fmt::Debug::fmt(self.destination.id(), f))
78            .and_then(|_| {
79                write!(
80                    f,
81                    ", tracker: {:?} }}",
82                    self.tracker
83                        .iter()
84                        .map(|n| format!("{:?}", n.node.id()))
85                        .collect::<Vec<_>>()
86                        .join("->")
87                )
88            })
89    }
90}

92impl<'s, K, D, N> Dfs<'s, K, D, N>
93where
94    K: Debug + Clone + Eq + Hash + 's,
95    D: Zero + Ord + Clone + Debug,
96    N: traits::IsNodeWithIndexedNeighbours<'s, K, D>,
97{
98    pub fn new(
99        start: &'s N,
100        destination: &'s N,
101        size_hint: usize,
102    ) -> Result<Self, SimpleGraphError<K, D>> {
103        if start.id() == destination.id() {
104            return Err(SimpleGraphError::CannotPathToSelf {
105                node: start.id().clone(),
106            });
107        }
108
109        let mut tracker = Vec::with_capacity(size_hint);
110        tracker.push(NodeInProgress::new(start, D::zero()));
111
112        Ok(Self {
113            start,
114            destination,
115            tracker,
116        })
117    }
118
119    #[allow(unused_assignments)]
120    pub fn next_solution(
121        &mut self,
122        get_node_by_key: impl Fn(&K) -> Option<&'s N> + Clone,
123    ) -> Option<(Vec<&'s K>, D)> {
124        while self.tracker.len() > 0 {
125            let opt_next_node = {
126                self.tracker
127                    .last_mut()
128                    .expect("Unreachable; memo length checked above")
129                    .next_unvisited_neighbour(get_node_by_key.clone())
130            };
131
132            match opt_next_node {
133                Some(next_node) => {
134                    if next_node.node.id() == self.destination.id() {
135                        let path_to_node = self
136                            .tracker
137                            .iter()
138                            .chain(std::iter::once(&next_node))
139                            .map(|n| n.node.id())
140                            .collect::<Vec<&'s K>>();
141
142                        #[cfg(feature = "trace")]
143                        eprintln!(
144                            "Found solution at node {:?} with distance {:?} and path {:?}",
145                            next_node.node.id(),
146                            next_node.distance,
147                            path_to_node
148                        );
149
150                        return Some((path_to_node, next_node.distance));
151                    } else {
152                        #[cfg(feature = "trace")]
153                        {
154                            let path_to_node = self
155                                .tracker
156                                .iter()
157                                .chain(std::iter::once(&next_node))
158                                .map(|n| n.node.id())
159                                .collect::<Vec<&'s K>>();
160                            eprintln!(
161                                "Visiting node {:?} with distance {:?} and path {:?}",
162                                next_node.node.id(),
163                                next_node.distance,
164                                path_to_node,
165                            );
166                        }
167                        self.tracker.push(next_node);
168                    }
169                }
170                None => {
171                    // Backtrack
172                    self.tracker.pop();
173                }
174            }
175        }
176
177        None
178    }
179}

181#[cfg(feature = "dfs-count")]
182pub fn dfs_count<'s, K, D, N>(
183    start: &'s N,
184    destination_id: &'s K,
185    size_hint: usize,
186    get_node_by_key: impl Fn(&K) -> Option<&'s N> + Clone,
187) -> usize
188where
189    K: Debug + Clone + Eq + Hash + 's,
190    D: Zero + Ord + Clone + Debug,
191    N: traits::IsNodeWithIndexedNeighbours<'s, K, D>,
192{
193    let mut count = 0;
194    let mut memoized_counts_by_node: fxhash::FxHashMap<&'s K, usize> =
195        FxHashMap::with_capacity_and_hasher(size_hint, Default::default());
196    let mut tracker: Vec<NodeInProgress<'s, K, D, N>> = Vec::with_capacity(size_hint);
197    tracker.push(NodeInProgress::new(start, D::zero()));
198
199    while tracker.len() > 0 {
200        let opt_next_node = {
201            tracker
202                .last_mut()
203                .expect("Unreachable; memo length checked above")
204                .next_unvisited_neighbour(get_node_by_key.clone())
205        };
206
207        match opt_next_node {
208            Some(next_node) if destination_id == next_node.node.id() => {
209                // We found a solution
210                count += 1;
211
212                #[cfg(feature = "trace")]
213                {
214                    let path_to_node = tracker
215                        .iter()
216                        .chain(std::iter::once(&next_node))
217                        .map(|n| n.node.id())
218                        .collect::<Vec<&'s K>>();
219
220                    eprintln!(
221                        "Found solution at node {:?} with distance {:?} and path {:?}",
222                        next_node.node.id(),
223                        next_node.distance,
224                        path_to_node
225                    );
226                }
227
228                let last_node = tracker
229                    .last()
230                    .expect("Unreachable; memo length checked above");
231
232                // If we have just reached a destination, the count increase is always 1 for the last node
233                memoized_counts_by_node
234                    .entry(last_node.node.id())
235                    .and_modify(|c| *c += 1)
236                    .or_insert(1);
237            }
238            Some(next_node) if memoized_counts_by_node.contains_key(next_node.node.id()) => {
239                // We have already computed the number of paths from this node to the destination
240                let unique_paths_from_next_node = *memoized_counts_by_node
241                    .get(next_node.node.id())
242                    .expect("Unreachable; checked above");
243
244                #[cfg(feature = "trace")]
245                {
246                    let path_to_node = tracker
247                        .iter()
248                        .chain(std::iter::once(&next_node))
249                        .map(|n| n.node.id())
250                        .collect::<Vec<&'s K>>();
251
252                    eprintln!(
253                        "Using memoized count {:?} at node {:?} with distance {:?} and path {:?}",
254                        unique_paths_from_next_node,
255                        next_node.node.id(),
256                        next_node.distance,
257                        path_to_node
258                    );
259                }
260
261                count += unique_paths_from_next_node;
262
263                let last_node = tracker
264                    .last()
265                    .expect("Unreachable; memo length checked above");
266                memoized_counts_by_node
267                    .entry(last_node.node.id())
268                    .and_modify(|c| *c += unique_paths_from_next_node)
269                    .or_insert(unique_paths_from_next_node);
270            }
271            Some(next_node) => {
272                #[cfg(feature = "trace")]
273                {
274                    let path_to_node = tracker
275                        .iter()
276                        .chain(std::iter::once(&next_node))
277                        .map(|n| n.node.id())
278                        .collect::<Vec<&'s K>>();
279
280                    eprintln!(
281                        "Visiting node {:?} with distance {:?} and path {:?}, no memoized count found",
282                        next_node.node.id(),
283                        next_node.distance,
284                        path_to_node,
285                    );
286                }
287                tracker.push(next_node);
288            }
289            None => {
290                // Backtrack
291                let _popped = tracker.pop().expect("Unreachable; memo length checked above");
292                
293                #[cfg(feature = "trace")]
294                {
295                    let path_to_node = tracker
296                    .iter()
297                    .map(|n| n.node.id())
298                    .collect::<Vec<&'s K>>();
299                    eprintln!("Backtracking from node {:?} to path {:?}", _popped.node.id(), path_to_node);
300                }
301            
302                if tracker.len() > 0 {
303                    // We should update the memoization for the last node in the tracker, even if the count is zero
304                    let count_from_popped = memoized_counts_by_node.get(&_popped.node.id()).copied().unwrap_or_default();
305                    let last_node = tracker
306                        .last()
307                        .expect("Unreachable; memo length checked above");
308                    memoized_counts_by_node
309                        .entry(last_node.node.id())
310                        .and_modify(|c| *c += count_from_popped)
311                        .or_insert(count_from_popped);
312                }
313            }
314        }
315    }
316
317    #[cfg(feature = "trace")]
318    eprintln!("Final memoized counts: {:?}", memoized_counts_by_node);
319
320    count
321}

330    fn test_dfs() {
331        let nodes: HashMap<u8, TestNode> = (1..=6)
332            .map(|id| (id, TestNode::new_with_connections(id, CONNECTIONS)))
333            .collect();
334
335        let start_node = nodes.get(&1).expect("Start node not found");
336        let destination_id = 5;
337        let get_node_by_key = |key: &u8| nodes.get(key);
338        let mut dfs = Dfs::new(
339            start_node,
340            get_node_by_key(&destination_id).expect("Unreachable, destination node not found"),
341            nodes.len(),
342        )
343        .expect("Failed to create DFS instance");
344
345        let solutions = {
346            let mut sols = HashSet::new();
347            while let Some(solution) = dfs.next_solution(get_node_by_key) {
348                sols.insert((
349                    solution.0.into_iter().map(|k| *k).collect::<Vec<u8>>(),
350                    solution.1,
351                ));
352            }
353            sols
354        };
355
356        let expected_solutions: HashSet<(Vec<u8>, u32)> = HashSet::from_iter(
357            [
358                (vec![1, 6, 5], 23),
359                (vec![1, 3, 4, 5], 26),
360                (vec![1, 3, 6, 5], 20),
361                (vec![1, 2, 4, 5], 28),
362                (vec![1, 2, 3, 6, 5], 28),
363                (vec![1, 2, 3, 4, 5], 34),
364            ]
365            .into_iter(),
366        );
367
368        assert_eq!(solutions, expected_solutions);
369    }
