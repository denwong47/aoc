7fn set_shape_in_storage(
8    bits: &mut StateStorage,
9    container: &Container,
10    shape: &Shape,
11    x: usize,
12    y: usize,
13) -> anyhow::Result<()> {
14    for dy in 0..shape.height() {
15        for dx in 0..shape.width() {
16            if shape.get(dx, dy) {
17                let index = (y + dy) * container.width + (x + dx);
18                if index >= container.size() {
19                    anyhow::bail!(
20                        "Index {index} for ({x}+{dx}, {y}+{dy}) out of bounds for container size {}",
21                        container.size()
22                    );
23                }
24                bits.set(index, true);
25            }
26        }
27    }
28    Ok(())
29}

76pub struct Placement<'r, const S: usize> {
77    pub shape_index: usize,
78    // This reference is for [`std::fmt::Display`] implementation only
79    pub requirement: &'r Requirement<S>,
80    pub x: usize,
81    pub y: usize,
82    pub shape_count: usize,
83    state: StateStorage,
84}
85
86impl<'r, const S: usize> Placement<'r, S> {
87    pub fn new<'s>(
88        shape: &'s Shape,
89        requirement: &'r Requirement<S>,
90        x: usize,
91        y: usize,
92        shape_count: usize,
93    ) -> anyhow::Result<Option<Self>> {
94        if let Some(shape_instance_offset) = requirement
95            .shape_counts
96            .get_shape_instance_offset(shape.index, shape_count)
97        {
98            let mut bits = requirement.build_new_state_storage();
99
100            // Set the bit for this shape instance
101            bits.set(requirement.container.size() + shape_instance_offset, true);
102
103            // Set the bits for the shape's position in the container
104            set_shape_in_storage(&mut bits, &requirement.container, shape, x, y)?;
105
106            Ok(Some(Self {
107                shape_index: shape.index,
108                requirement,
109                x,
110                y,
111                shape_count,
112                state: bits,
113            }))
114        } else {
115            Ok(None)
116        }
117    }
118
119    pub fn state(&self) -> &StateStorage {
120        &self.state
121    }
122
123    /// Check if the placement fills the container at (x, y)
124    pub fn is_filled_at(&self, x: usize, y: usize) -> bool {
125        let width = self.requirement.container.width;
126        x < width
127            && y < self.requirement.container.height
128            && self.state.get(y * width + x).unwrap_or(false)
129    }
130
131    /// Check if the shape instance is set in the placement
132    pub fn is_shape_instance_set(&self, shape_index: usize, shape_count: usize) -> bool {
133        if let Some(offset) = self
134            .requirement
135            .shape_counts
136            .get_shape_instance_offset(shape_index, shape_count)
137        {
138            let index = self.requirement.container.size() + offset;
139            self.state.get(index).unwrap_or(false)
140        } else {
141            false
142        }
143    }
144}

158/// Pre-compute all possible placements of shapes within the requirement's container.
159pub fn build_placements_for_requirement<'r, const S: usize>(
160    shapes: &[Shape],
161    requirement: &'r Requirement<S>,
162) -> Vec<Placement<'r, S>> {
163    let total_placements_count = progress::calculate_total_placements(shapes, requirement);
164    tqdm!(
165        (0..shapes.len())
166            .cartesian_product(
167                requirement
168                    .container
169                    .iter_all_positions(shapes[0].width(), shapes[0].height())
170            )
171            .flat_map(|(shape_index, (x, y))| {
172                PlacementBuilder::new(&shapes[shape_index], requirement, x, y)
173            }),
174        // Set the total count for the progress bar, part of the `tqdm!()` macro
175        total = total_placements_count
176    )
177    .inspect(|placement| {
178        assert!(
179            !placement.state().is_empty(),
180            "Generated placement has empty state!"
181        );
182    })
183    .collect::<Vec<_>>()
184}
