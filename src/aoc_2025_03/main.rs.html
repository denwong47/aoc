76fn build_vec_deque_from_string(s: &str) -> VecDeque<u8> {
77    VecDeque::from(s.chars().map(|c| c as u8 - b'0').collect::<Vec<u8>>())
78}
79trait HighestSequentialCombination {
80    fn filter_to_highest_sequential_combination<const N: usize>(&mut self);
81    fn build_int(&self) -> u64;
82}
83
84impl HighestSequentialCombination for VecDeque<u8> {
85    fn filter_to_highest_sequential_combination<const N: usize>(&mut self) {
86        let mut pos: usize = 0;
87        loop {
88            if pos == self.len() - 1 || self.len() <= N {
89                break;
90            }
91            if self[pos] < self[pos + 1] {
92                self.remove(pos);
93                pos = pos.saturating_sub(1);
94            } else {
95                pos += 1;
96            }
97        }
98
99        if self.len() > N {
100            drop(self.split_off(N));
101        }
102    }
103
104    fn build_int(&self) -> u64 {
105        self.iter()
106            .fold(0u64, |acc, &digit| acc * 10u64 + digit as u64)
107    }
108}

110fn main() {
111    let sum2: u64 = INPUT
112        .split_whitespace()
113        .map(|line| {
114            let mut vec_deque = build_vec_deque_from_string(line);
115            vec_deque.filter_to_highest_sequential_combination::<2>();
116            vec_deque.build_int()
117        })
118        .sum();
119    let sum12: u64 = INPUT
120        .split_whitespace()
121        .map(|line| {
122            let mut vec_deque = build_vec_deque_from_string(line);
123            vec_deque.filter_to_highest_sequential_combination::<12>();
124            vec_deque.build_int()
125        })
126        .sum();
127
128    println!("Part 1 Total Joltage: \x1b[32m{}\x1b[0m", sum2);
129    println!("Part 2 Total Joltage: \x1b[32m{}\x1b[0m", sum12);
130}
