5/// Hash a value into a state where it can be wrapping added or removed from the
6/// accumulative hash.
7pub fn hash<T: IsAccumulativeHashType + From<S>, S>(value: S) -> T {
8    let mut z = (T::from(value)).wrapping_add(&T::SEED);
9    z = (z ^ (z >> T::SHIFT_CONSTANTS[0])).wrapping_mul(&T::MULTIPLIER_CONSTANTS[0]);
10    z = (z ^ (z >> T::SHIFT_CONSTANTS[1])).wrapping_mul(&T::MULTIPLIER_CONSTANTS[1]);
11    z ^ (z >> T::SHIFT_CONSTANTS[2])
12}
