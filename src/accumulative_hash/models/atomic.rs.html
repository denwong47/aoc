1//! The standard model for accumulative hashing, without atomic types.
2
3use crate::{AccumulativeHash, IsAtomicAccumulativeHashType, helpers};
4
5use std::sync::atomic::Ordering;
6
7use num_traits::{WrappingAdd, WrappingSub, Zero};
8
9/// A struct that remembers the state of a hash as data is added and/or removed from it.
10#[derive(Debug)]
11pub struct AtomicAccumulativeHash<T: IsAtomicAccumulativeHashType> {
12    state: T,
13}
14
15impl<T: IsAtomicAccumulativeHashType> Clone for AtomicAccumulativeHash<T> {
16    /// Create a clone of the current atomic accumulative hash.
17    ///
18    /// This performs a relaxed load of the current state to initialize the new instance.
19    ///
20    /// The new instance is cloned by value, and does not share state with the original instance.
21    fn clone(&self) -> Self {
22        Self::with_state(self.state.to_underlying(Ordering::Relaxed))
23    }
24}
25
26impl<T: IsAtomicAccumulativeHashType> AtomicAccumulativeHash<T> {
27    /// Create a new empty accumulative hash.
28    ///
29    /// The initial state is equivalent to hashing no values.
30    pub fn new() -> Self {
31        Self::with_state(T::UnderlyingType::zero().into())
32    }
33
34    /// Internal method to add a hashed value to the current state atomically.
35    pub fn _raw_op(
36        &self,
37        is_add: bool,
38        hashed_value: T::UnderlyingType,
39        success: Ordering,
40        failure: Ordering,
41    ) -> T::UnderlyingType {
42        let mut current_state = self.load(failure);
43        loop {
44            let new_state = if is_add {
45                current_state.wrapping_add(&hashed_value)
46            } else {
47                current_state.wrapping_sub(&hashed_value)
48            };
49
50            match self.state.compare_exchange(
51                current_state.into(),
52                new_state.into(),
53                success,
54                failure,
55            ) {
56                // If the exchange was successful, return the new state, and discard the previous state.
57                Ok(_) => return new_state,
58                // If the exchange failed, update current_state and retry
59                Err(actual) => current_state = actual.into(),
60            }
61        }
62    }
63
64    /// Hash a value and combine it with the current state, returning the new hash state,
65    /// but not modifying the internal state.
66    /// 
67    /// This is useful for checking what the hash would be if a value were to be added,
68    /// without actually modifying the accumulative hash.
69    pub fn and_hash<S: Into<T::UnderlyingType>>(&self, value: S, order: Ordering) -> T::UnderlyingType {
70        let value_as_underlying = value.into();
71        let hashed = helpers::hash::<T::UnderlyingType, _>(value_as_underlying);
72        self.load(order).wrapping_add(&hashed)
73    }
74
75
76    /// Add a value to the accumulative hash.
77    ///
78    /// This does not guarantee that the value was never added before; it will simply
79    /// add the hashed value to the current state.
80    ///
81    /// This means that adding the same value multiple times will affect the hash state
82    /// accordingly.
83    ///
84    /// This method uses atomic operations to ensure thread safety; it takes two
85    /// [`Ordering`] parameters to specify the memory ordering for success and failure cases.
86    /// ``failure`` is also used when loading the current state for the first time.
87    ///
88    /// It guarantees that the returned state is the result of the addition, and no race
89    /// conditions can occur, at the cost of potential retries if other threads modify
90    /// the state concurrently.
91    pub fn add<S: Into<T::UnderlyingType>>(
92        &self,
93        value: S,
94        success: Ordering,
95        failure: Ordering,
96    ) -> T::UnderlyingType {
97        let value_as_underlying = value.into();
98        let hashed = helpers::hash::<T::UnderlyingType, _>(value_as_underlying);
99
100        self._raw_op(true, hashed, success, failure)
101    }
102
103    /// Remove a value from the accumulative hash.
104    ///
105    /// This does not guarantee that the value was previously added;
106    /// it will simply subtract the hashed value from the current state.
107    ///
108    /// This means that removing a value that was never added may lead to
109    /// undetermined behavior; it can be fixed by re-adding the value later,
110    /// but the intermediate state may not be valid.
111    ///
112    /// This method uses atomic operations to ensure thread safety; it takes two
113    /// [`Ordering`] parameters to specify the memory ordering for success and failure cases.
114    /// ``failure`` is also used when loading the current state for the first time.
115    ///
116    /// It guarantees that the returned state is the result of the removal, and no race
117    /// conditions can occur, at the cost of potential retries if other threads modify
118    /// the state concurrently.
119    pub fn remove<S: Into<T::UnderlyingType>>(
120        &self,
121        value: S,
122        success: Ordering,
123        failure: Ordering,
124    ) -> T::UnderlyingType {
125        let value_as_underlying = value.into();
126        let hashed = helpers::hash::<T::UnderlyingType, _>(value_as_underlying);
127
128        self._raw_op(false, hashed, success, failure)
129    }
130
131    /// Add multiple values to the accumulative hash.
132    ///
133    /// This does not guarantee that the values were never added before; it will simply
134    /// add the hashed values to the current state.
135    ///
136    /// This means that adding the same value multiple times will affect the hash state
137    /// accordingly.
138    ///
139    /// This method uses atomic operations to ensure thread safety; it takes two
140    /// [`Ordering`] parameters to specify the memory ordering for success and failure cases.
141    /// ``failure`` is also used when loading the current state for the first time.
142    ///
143    /// It guarantees that the returned state is the result of the addition, and no race
144    /// conditions can occur, at the cost of potential retries if other threads modify
145    /// the state concurrently.
146    pub fn add_multiple<S: Into<T::UnderlyingType>, I: IntoIterator<Item = S>>(
147        &self,
148        values: I,
149        success: Ordering,
150        failure: Ordering,
151    ) -> T::UnderlyingType {
152        // Pre-calculate the combined hash of all values to added first, so that we can reduce the race window
153        // between loading the current state and updating it.
154        let combined_state = AccumulativeHash::<T::UnderlyingType>::from(values).into_state();
155
156        self._raw_op(true, combined_state, success, failure)
157    }
158
159    /// Remove multiple values from the accumulative hash.
160    ///
161    /// This does not guarantee that the values were previously added;
162    /// it will simply subtract the hashed values from the current state.
163    ///
164    /// This means that removing values that were never added may lead to
165    /// undetermined behavior; it can be fixed by re-adding the values later,
166    /// but the intermediate state may not be valid.
167    ///
168    /// This method uses atomic operations to ensure thread safety; it takes two
169    /// [`Ordering`] parameters to specify the memory ordering for success and failure cases.
170    /// ``failure`` is also used when loading the current state for the first time.
171    ///
172    /// It guarantees that the returned state is the result of the removal, and no race
173    /// conditions can occur, at the cost of potential retries if other threads modify
174    /// the state concurrently.
175    pub fn remove_multiple<S: Into<T::UnderlyingType>, I: IntoIterator<Item = S>>(
176        &self,
177        values: I,
178        success: Ordering,
179        failure: Ordering,
180    ) -> T::UnderlyingType {
181        // Pre-calculate the combined hash of all values to removed first, so that we can reduce the race window
182        // between loading the current state and updating it.
183        let combined_state = AccumulativeHash::<T::UnderlyingType>::from(values).into_state();
184
185        self._raw_op(false, combined_state, success, failure)
186    }
187
188    /// Create a new accumulative hash with an initial state.
189    pub fn with_state(state: T::UnderlyingType) -> Self {
190        Self {
191            state: state.into(),
192        }
193    }
194
195    /// Get the current state of the accumulative hash.
196    pub fn state(&self) -> &T {
197        &self.state
198    }
199
200    /// Load the current state of the accumulative hash as the underlying type.
201    pub fn load(&self, order: Ordering) -> T::UnderlyingType {
202        self.state.to_underlying(order)
203    }
204
205    /// Extend the current accumulative hash with another accumulative hash.
206    pub fn extend(&self, other: &Self, success: Ordering, failure: Ordering) {
207        let other_state = other.load(failure);
208
209        self._raw_op(true, other_state, success, failure);
210    }
211}

242                #[test]
243                fn sequential_add_must_equal_multiple_add() {
244                    let acc_hash_seq = AtomicAccumulativeHash::<$typ>::new();
245                    for &value in SEQUENCE_TO_ADD_1.iter() {
246                        acc_hash_seq.add(value, STORE_ORDER, LOAD_ORDER);
247                    }
248                    let state_seq = acc_hash_seq.load(LOAD_ORDER);
249
250                    let acc_hash_multi = AtomicAccumulativeHash::<$typ>::new();
251                    acc_hash_multi.add_multiple(SEQUENCE_TO_ADD_1.iter().cloned(), STORE_ORDER, LOAD_ORDER);
252                    let state_multi = acc_hash_multi.load(LOAD_ORDER);
253
254                    assert_eq!(state_seq, state_multi, "Sequential add and multiple add states do not match.");

311                #[test]
312                fn removing_values_must_return_to_original_state() {
313                    let acc_hash_1 = AtomicAccumulativeHash::<$typ>::new();
314
315                    acc_hash_1.add_multiple(SEQUENCE_TO_ADD_1.iter().cloned(), STORE_ORDER, LOAD_ORDER);
316                    acc_hash_1.remove_multiple(SEQUENCE_TO_REMOVE_1.iter().cloned(), STORE_ORDER, LOAD_ORDER);
317                    acc_hash_1.add_multiple(SEQUENCE_TO_ADD_2.iter().cloned(), STORE_ORDER, LOAD_ORDER);
318                    acc_hash_1.remove_multiple(SEQUENCE_TO_REMOVE_2.iter().cloned(), STORE_ORDER, LOAD_ORDER);
319
320                    let mut combined_values: HashSet<u8> = HashSet::from_iter(SEQUENCE_TO_ADD_1.iter().cloned());
321                    SEQUENCE_TO_REMOVE_1.iter().for_each(|&v| { combined_values.remove(&v); });
322                    SEQUENCE_TO_ADD_2.iter().for_each(|&v| { combined_values.insert(v); });
323                    SEQUENCE_TO_REMOVE_2.iter().for_each(|&v| { combined_values.remove(&v); });
324                    dbg!(&combined_values);
325
326                    let acc_hash_2 = AtomicAccumulativeHash::<$typ>::new();
327                    acc_hash_2.add_multiple(combined_values.iter().cloned(), STORE_ORDER, LOAD_ORDER);
328
329                    assert_eq!(acc_hash_1.load(LOAD_ORDER), acc_hash_2.load(LOAD_ORDER), "States do not match after adding and removing sequences versus combined operations.");
330                }

332                #[test]
333                fn merging_states_must_equal_individual_operations() {
334                    let acc_hash_1 = AtomicAccumulativeHash::<$typ>::new();
335                    acc_hash_1.add_multiple(SEQUENCE_TO_ADD_1.iter().cloned(), STORE_ORDER, LOAD_ORDER);
336                    acc_hash_1.remove_multiple(SEQUENCE_TO_REMOVE_1.iter().cloned(), STORE_ORDER, LOAD_ORDER);
337
338                    let  acc_hash_2 = AtomicAccumulativeHash::<$typ>::new();
339                    acc_hash_2.add_multiple(SEQUENCE_TO_ADD_2.iter().cloned(), STORE_ORDER, LOAD_ORDER);
340                    acc_hash_2.remove_multiple(SEQUENCE_TO_REMOVE_2.iter().cloned(), STORE_ORDER, LOAD_ORDER);
341
342                    acc_hash_1.extend(&acc_hash_2, STORE_ORDER, LOAD_ORDER);
343
344                    let individual_acc_hash = AtomicAccumulativeHash::<$typ>::new();
345                    individual_acc_hash.add_multiple(SEQUENCE_TO_ADD_1.iter().cloned(), STORE_ORDER, LOAD_ORDER);
346                    individual_acc_hash.remove_multiple(SEQUENCE_TO_REMOVE_1.iter().cloned(), STORE_ORDER, LOAD_ORDER);
347                    individual_acc_hash.add_multiple(SEQUENCE_TO_ADD_2.iter().cloned(), STORE_ORDER, LOAD_ORDER);
348                    individual_acc_hash.remove_multiple(SEQUENCE_TO_REMOVE_2.iter().cloned(), STORE_ORDER, LOAD_ORDER);
349
350                    assert_eq!(acc_hash_1.load(LOAD_ORDER), individual_acc_hash.load(LOAD_ORDER), "Merged state does not equal individual operations state.");
351                }
352            }
353        };
