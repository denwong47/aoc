1//! The standard model for accumulative hashing, without atomic types.
2
3use crate::{IsAccumulativeHashType, helpers};
4
5/// A struct that remembers the state of a hash as data is added and/or removed from it.
6///
7/// The order of the item is NOT considered when calculating the hash --
8/// ``A-B-C`` will have the same hash as ``C-B-A``. This is deliberate, and useful for
9/// addictive data structures that need to check for equality regardless of order,
10/// where traditional hashing requires sorting beforehand.
11#[derive(Debug, Clone, Copy, PartialEq, Eq)]
12pub struct AccumulativeHash<T: IsAccumulativeHashType> {
13    state: T,
14}
15
16impl<T: IsAccumulativeHashType> AccumulativeHash<T> {
17    /// Create a new empty accumulative hash.
18    ///
19    /// The initial state is equivalent to hashing no values.
20    pub fn new() -> Self {
21        Self::with_state(T::zero())
22    }
23
24    /// Create a new accumulative hash with an initial state.
25    pub fn with_state(state: T) -> Self {
26        Self { state }
27    }
28
29    /// Hash a value and combine it with the current state, returning the new hash state,
30    /// but not modifying the internal state.
31    /// 
32    /// This is useful for checking what the hash would be if a value were to be added,
33    /// without actually modifying the accumulative hash.
34    pub fn and_hash<S: Into<T>>(&self, value: S) -> T {
35        let hashed = helpers::hash::<T, _>(value.into());
36        self.state.wrapping_add(&hashed)
37    }
38
39    /// Add a value to the accumulative hash.
40    ///
41    /// This does not guarantee that the value was never added before;
42    /// it will simply add the hashed value to the current state.
43    ///
44    /// This means that adding the same value multiple times will
45    /// affect the hash state accordingly.
46    pub fn add<S: Into<T>>(&mut self, value: S) -> &T {
47        let hashed = helpers::hash::<T, _>(value.into());
48        self.state = self.state.wrapping_add(&hashed);
49
50        self.state()
51    }
52
53    /// Remove a value from the accumulative hash.
54    ///
55    /// This does not guarantee that the value was previously added;
56    /// it will simply subtract the hashed value from the current state.
57    ///
58    /// This means that removing a value that was never added may lead to
59    /// undetermined behavior; it can be fixed by re-adding the value later,
60    /// but the intermediate state may not be valid.
61    pub fn remove<S: Into<T>>(&mut self, value: S) -> &T {
62        let hashed = helpers::hash::<T, _>(value.into());
63        self.state = self.state.wrapping_sub(&hashed);
64
65        self.state()
66    }
67
68    /// Add multiple values to the accumulative hash.
69    pub fn add_multiple<S: Into<T>, I: IntoIterator<Item = S>>(&mut self, values: I) -> &T {
70        for value in values {
71            self.add(value);
72        }
73        self.state()
74    }
75
76    /// Remove multiple values from the accumulative hash.
77    pub fn remove_multiple<S: Into<T>, I: IntoIterator<Item = S>>(&mut self, values: I) -> &T {
78        for value in values {
79            self.remove(value);
80        }
81        self.state()
82    }
83
84    /// Get the current state of the accumulative hash.
85    pub fn state(&self) -> &T {
86        &self.state
87    }
88
89    /// Extend this accumulative hash by merging another accumulative hash into it.
90    ///
91    /// Hashing in this way guarantees: ``hash([A]) + hash([B]) == hash([A, B])`` where
92    /// ``hash`` represents [`AccumulativeHash::add_multiple`] on different instances of
93    /// [`AccumulativeHash`].
94    ///
95    /// This allows us to combine two accumulative hashes into one, effectively merging
96    /// their states.
97    ///
98    /// Since ``T`` implements [`Copy`], we can afford to copy the state of the other
99    /// accumulative hash without worrying about cost.
100    pub fn extend(&mut self, other: &AccumulativeHash<T>) -> &T {
101        self.state = self.state.wrapping_add(&other.state);
102        &self.state
103    }
104
105    /// Consume this accumulative hash and return its current state.
106    pub fn into_state(self) -> T {
107        self.state
108    }
109}
110
111/// [`AccumulativeHash`] can be created from any iterable collection of values.
112impl<T: IsAccumulativeHashType, I> From<I> for AccumulativeHash<T>
113where
114    I: IntoIterator,
115    I::Item: Into<T>,
116{
117    /// Create an accumulative hash from an iterable collection of values.
118    fn from(value: I) -> Self {
119        let mut acc_hash = AccumulativeHash::<T>::new();
120        acc_hash.add_multiple(value);
121        acc_hash
122    }
123}

125#[cfg(test)]
126mod tests {
127    use super::*;
128
129    const SEQUENCE_TO_ADD_1: &'static [u8] = &[1, 2, 4, 8, 16, 32, 64, 128];
130    const SEQUENCE_TO_REMOVE_1: &'static [u8] = &[1, 4, 8, 64];
131    const SEQUENCE_TO_ADD_2: &'static [u8] = &[3, 6, 9, 12, 15];
132    const SEQUENCE_TO_REMOVE_2: &'static [u8] = &[2, 6, 12];
133
134    macro_rules! test_type {
135        ($name:ident::<$typ:ident>(add_1=$add_1:literal, remove_1=$remove_1:literal, add_2=$add_2:literal, remove_2=$remove_2:literal)) => {
136            mod $name {
137                use super::*;
138                use num_traits::Zero;
139                use std::collections::HashSet;
140
141                #[test]
142                fn sequential_add_must_equal_multiple_add() {
143                    let mut acc_hash_seq = AccumulativeHash::<$typ>::new();
144                    for &value in SEQUENCE_TO_ADD_1.iter() {
145                        acc_hash_seq.add(value);
146                    }
147                    let state_seq = *acc_hash_seq.state();
148
149                    let mut acc_hash_multi = AccumulativeHash::<$typ>::new();
150                    acc_hash_multi.add_multiple(SEQUENCE_TO_ADD_1.iter().cloned());
151                    let state_multi = *acc_hash_multi.state();
152
153                    assert_eq!(state_seq, state_multi, "Sequential add and multiple add states do not match.");
154                }
155
156                #[test]
157                fn sequential_add_must_equal_to_unordered_add() {
158                    let mut acc_hash_seq = AccumulativeHash::<$typ>::new();
159                    let state_seq = *acc_hash_seq.add_multiple(SEQUENCE_TO_ADD_1.iter().cloned());
160
161                    let mut acc_hash_unordered = AccumulativeHash::<$typ>::new();
162                    acc_hash_unordered.add_multiple(SEQUENCE_TO_ADD_1.iter().rev().cloned());
163                    let state_unordered = *acc_hash_unordered.state();
164
165                    assert_eq!(state_seq, state_unordered, "Sequential add and unordered add states do not match.");
166                }
167
168                #[test]
169                fn sequential_remove_must_equal_multiple_remove() {
170                    let mut acc_hash_seq = AccumulativeHash::<$typ>::new();
171                    acc_hash_seq.add_multiple(SEQUENCE_TO_ADD_1.iter().cloned());
172
173                    let mut acc_hash_unordered = acc_hash_seq.clone();
174
175                    let state_seq = *acc_hash_seq.remove_multiple(SEQUENCE_TO_REMOVE_1.iter().cloned());
176                    let state_multi = *acc_hash_unordered.remove_multiple(SEQUENCE_TO_REMOVE_1.iter().rev().cloned());
177
178                    assert_eq!(state_seq, state_multi, "Sequential remove and multiple remove states do not match.");
179                }
180
181                #[test]
182                fn sequential_add_and_remove() {
183                    let mut acc_hash = AccumulativeHash::<$typ>::new();
184
185                    let expected_after_add_1 = $add_1 as $typ;
186                    let actual_after_add_1 = *acc_hash.add_multiple(SEQUENCE_TO_ADD_1.iter().cloned());
187                    assert!(actual_after_add_1 == expected_after_add_1, "Hashes not matching after first addition: \x1b[31moutput\x1b[0m \u{2192} \x1b[34m\x1b[1m0x{:X}\x1b[22m {} \x1b[1m0x{:X}\x1b[0m \u{2190} \x1b[32mexpected\x1b[0m", actual_after_add_1, stringify!(==), expected_after_add_1);
188
189                    let expected_after_remove_1 = $remove_1 as $typ;
190                    let actual_after_remove_1 = *acc_hash.remove_multiple(SEQUENCE_TO_REMOVE_1.iter().cloned());
191                    assert!(actual_after_remove_1 == expected_after_remove_1, "Hashes not matching after first removal: \x1b[31moutput\x1b[0m \u{2192} \x1b[34m\x1b[1m0x{:X}\x1b[22m {} \x1b[1m0x{:X}\x1b[0m \u{2190} \x1b[32mexpected\x1b[0m", actual_after_remove_1, stringify!(==), expected_after_remove_1);
192                    let expected_after_add_2 = $add_2 as $typ;
193                    let actual_after_add_2 = *acc_hash.add_multiple(SEQUENCE_TO_ADD_2.iter().cloned());
194                    assert!(actual_after_add_2 == expected_after_add_2, "Hashes not matching after second addition: \x1b[31moutput\x1b[0m \u{2192} \x1b[34m\x1b[1m0x{:X}\x1b[22m {} \x1b[1m0x{:X}\x1b[0m \u{2190} \x1b[32mexpected\x1b[0m", actual_after_add_2, stringify!(==), expected_after_add_2);
195
196                    let expected_after_remove_2 = $remove_2 as $typ;
197                    let actual_after_remove_2 = *acc_hash.remove_multiple(SEQUENCE_TO_REMOVE_2.iter().cloned());
198                    assert!(actual_after_remove_2 == expected_after_remove_2, "Hashes not matching after second removal: \x1b[31moutput\x1b[0m \u{2192} \x1b[34m\x1b[1m0x{:X}\x1b[22m {} \x1b[1m0x{:X}\x1b[0m \u{2190} \x1b[32mexpected\x1b[0m", actual_after_remove_2, stringify!(==), expected_after_remove_2);
199                }

201                #[test]
202                fn adding_and_removing_same_values_must_return_to_initial_state() {
203                    let mut acc_hash = AccumulativeHash::<$typ>::new();
204
205                    acc_hash.add_multiple(SEQUENCE_TO_ADD_1.iter().cloned());
206                    acc_hash.remove_multiple(SEQUENCE_TO_ADD_1.iter().cloned());
207                    assert_eq!(*acc_hash.state(), $typ::zero(), "State after adding and removing the same values did not return to initial state.");
208                }
209
210                #[test]
211                fn removing_values_must_return_to_original_state() {
212                    let mut acc_hash_1 = AccumulativeHash::<$typ>::new();
213
214                    acc_hash_1.add_multiple(SEQUENCE_TO_ADD_1.iter().cloned());
215                    acc_hash_1.remove_multiple(SEQUENCE_TO_REMOVE_1.iter().cloned());
216                    acc_hash_1.add_multiple(SEQUENCE_TO_ADD_2.iter().cloned());
217                    acc_hash_1.remove_multiple(SEQUENCE_TO_REMOVE_2.iter().cloned());
218
219                    let mut combined_values: HashSet<u8> = HashSet::from_iter(SEQUENCE_TO_ADD_1.iter().cloned());
220                    SEQUENCE_TO_REMOVE_1.iter().for_each(|&v| { combined_values.remove(&v); });
221                    SEQUENCE_TO_ADD_2.iter().for_each(|&v| { combined_values.insert(v); });
222                    SEQUENCE_TO_REMOVE_2.iter().for_each(|&v| { combined_values.remove(&v); });
223                    dbg!(&combined_values);
224
225                    let mut acc_hash_2 = AccumulativeHash::<$typ>::new();
226                    acc_hash_2.add_multiple(combined_values.iter().cloned());
227
228                    assert_eq!(*acc_hash_1.state(), *acc_hash_2.state(), "States do not match after adding and removing sequences versus combined operations.");
229                }
230
231                #[test]
232                fn merging_states_must_equal_individual_operations() {
233                    let mut acc_hash_1 = AccumulativeHash::<$typ>::new();
234                    acc_hash_1.add_multiple(SEQUENCE_TO_ADD_1.iter().cloned());
235                    acc_hash_1.remove_multiple(SEQUENCE_TO_REMOVE_1.iter().cloned());
236
237                    let mut acc_hash_2 = AccumulativeHash::<$typ>::new();
238                    acc_hash_2.add_multiple(SEQUENCE_TO_ADD_2.iter().cloned());
239                    acc_hash_2.remove_multiple(SEQUENCE_TO_REMOVE_2.iter().cloned());
240
241                    acc_hash_1.extend(&acc_hash_2);
242
243                    let mut individual_acc_hash = AccumulativeHash::<$typ>::new();
244                    individual_acc_hash.add_multiple(SEQUENCE_TO_ADD_1.iter().cloned());
245                    individual_acc_hash.remove_multiple(SEQUENCE_TO_REMOVE_1.iter().cloned());
246                    individual_acc_hash.add_multiple(SEQUENCE_TO_ADD_2.iter().cloned());
247                    individual_acc_hash.remove_multiple(SEQUENCE_TO_REMOVE_2.iter().cloned());
248
249                    assert_eq!(*acc_hash_1.state(), *individual_acc_hash.state(), "Merged state does not equal individual operations state.");
250                }
251            }
252        };
253    }
