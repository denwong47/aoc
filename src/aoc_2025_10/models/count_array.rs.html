51impl CountArray<u16> {
52    pub fn mask(&self) -> CountArray<bool> {
53        CountArray {
54            values: self.values.iter().map(|v| (v & 1) == 1).collect(),
55        }
56    }
57
58    pub fn mut_add<T>(&mut self, other: &CountArray<T>) -> anyhow::Result<()>
59    where
60        T: Into<u16> + Clone + Copy + Debug + PartialEq + Eq,
61    {
62        self.values
63            .iter_mut()
64            .zip(other.values.iter())
65            .try_for_each(|(mine, yours)| {
66                *mine = mine
67                    .checked_add((*yours).into())
68                    .ok_or_else(|| anyhow::anyhow!("u16 overflowed during addition"))?;
69
70                Ok(())
71            })
72    }
73
74    pub fn add<T>(&self, other: &CountArray<T>) -> anyhow::Result<Self>
75    where
76        T: Into<u16> + Clone + Copy + Debug + PartialEq + Eq,
77    {
78        let mut new = self.clone();
79        new.mut_add(other)?;
80        Ok(new)
81    }
82
83    pub fn difference_from(&self, other: &Self) -> Difference<i16> {
84        Difference::from(
85            self.iter()
86                .zip(other.iter())
87                .map(|(mine, yours)| (*mine as i16) - (*yours as i16))
88                .collect::<Vec<i16>>(),
89        )
90    }
