3/// Finds the largest area from a list of indexed coordinates.
4///
5/// ``indexed_coords`` must be sorted in their index in ascending order.
6pub fn find_best_match(
7    indexed_coords: &[IndexedCoords],
8    predicate: impl Fn(&Rectangle, &Rectangle) -> anyhow::Result<std::cmp::Ordering>,
9) -> anyhow::Result<Option<Rectangle>> {
10    indexed_coords.iter().fold(
11        Ok(None),
12        |candidate: anyhow::Result<Option<Rectangle>>, &current| {
13            indexed_coords[current.index + 1..]
14                .iter()
15                .fold(candidate, |inner_candidate, &next| {
16                    if let Ok(opt_rec) = inner_candidate {
17                        let rect = Rectangle::new(current, next);
18                        #[cfg(feature = "trace")]
19                        {
20                            eprintln!(
21                                "Considering rectangle between {:?} and {:?} with area {}",
22                                current.coords,
23                                next.coords,
24                                rect.area()
25                            );
26                        }
27                        match opt_rec {
28                            Some(current_candidate) => {
29                                if rect.compare(&current_candidate, &predicate)?
30                                    == std::cmp::Ordering::Greater
31                                {
32                                    Ok(Some(rect))
33                                } else {
34                                    Ok(Some(current_candidate))
35                                }
36                            }
37                            None => Ok(Some(rect)),
38                        }
39                    } else {
40                        inner_candidate
41                    }
42                })
43        },
44    )
45}

47pub fn compare_area_with_visibility(
48    candidate: &Rectangle,
49    current: &Rectangle,
50) -> anyhow::Result<std::cmp::Ordering> {
51    let chosen_points = (
52        candidate.original_coords[0].index,
53        candidate.original_coords[1].index,
54    );
55    let target = (1, 30);
56    match candidate.area().cmp(&current.area()) {
57        std::cmp::Ordering::Greater => {
58            let results = candidate
59                .original_points_by_corners()
60                .map(|(corner, indexed_coords)| -> anyhow::Result<bool> {
61                    let visbounds = indexed_coords.visibility_bounds.ok_or(anyhow::anyhow!(
62                        "Indexed coordinate {:?} has no visibility bounds",
63                        indexed_coords.coords
64                    ))?;
65
66                    if chosen_points == target{
67                        dbg!(&corner, &indexed_coords);
68                    }
69
70                    let is_within = match corner {
71                        Corner::TopLeft => {
72                            // Check if the top left corner can see beyond the candidate rectangle
73                            visbounds
74                                .right
75                                .map(|right_bound| right_bound >= candidate.top_right()[0])
76                                .unwrap_or(false)
77                                && visbounds
78                                    .bottom
79                                    .map(|bottom_bound| {
80                                        bottom_bound >= candidate.bottom_left()[1]
81                                    })
82                                    .unwrap_or(false)
83                        }
84                        Corner::TopRight => {
85                            visbounds
86                                .left
87                                .map(|left_bound| left_bound <= candidate.top_left()[0])
88                                .unwrap_or(false)
89                                && visbounds
90                                    .bottom
91                                    .map(|bottom_bound| {
92                                        bottom_bound >= candidate.bottom_right()[1]
93                                    })
94                                    .unwrap_or(false)
95                        }
96                        Corner::BottomLeft => {
97                            visbounds
98                                .right
99                                .map(|right_bound| right_bound >= candidate.bottom_right()[0])
100                                .unwrap_or(false)
101                                && visbounds
102                                    .top
103                                    .map(|top_bound| {
104                                        top_bound <= candidate.top_left()[1]
105                                    })
106                                    .unwrap_or(false)
107                        }
108                        Corner::BottomRight => {
109                            visbounds
110                                .left
111                                .map(|left_bound| left_bound <= candidate.bottom_left()[0])
112                                .unwrap_or(false)
113                                && visbounds
114                                    .top
115                                    .map(|top_bound| top_bound <= candidate.top_right()[1])
116                                    .unwrap_or(false)
117                        }
118                    };
119
120                    if chosen_points == target {
121                        dbg!(&is_within);
122                    }
123
124                    #[cfg(feature = "trace")]
125                    {
126                        if is_within {
127                            eprintln!(
128                                "{:?} at {:?} can see the neighbouring corners of candidate rectangle",
129                                corner, indexed_coords.coords
130                            );
131                        } else {
132                            eprintln!(
133                                "{:?} at {:?} cannot see the neighbouring corners of candidate rectangle with bounds {:?}",
134                                corner, indexed_coords.coords, visbounds
135                            );
136                        }
137                    }
138
139                    Ok(is_within)
140                })
141                .collect::<anyhow::Result<Vec<_>>>()?;
142
143            if results.iter().all(|&v| v) {
144                #[cfg(feature = "trace")]
145                {
146                    eprintln!(
147                        "Candidate rectangle with area {} is bigger than current with area {} and is within the polygon",
148                        candidate.area(),
149                        current.area()
150                    );
151                }
152                Ok(std::cmp::Ordering::Greater)
153            } else {
154                #[cfg(feature = "trace")]
155                {
156                    eprintln!(
157                        "Candidate rectangle with area {} is bigger than current with area {} but is NOT within the polygon",
158                        candidate.area(),
159                        current.area()
160                    );
161                }
162                Ok(std::cmp::Ordering::Less)
163            }
164        }
165        ord => Ok(ord),
166    }
167}
