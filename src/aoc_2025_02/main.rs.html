13fn split_input_into_iterables(
14    input: &str,
15) -> impl Iterator<Item = impl Iterator<Item = u64> + '_> + '_ {
16    input.split(',').map(|section| {
17        let mut bounds = section
18            .split('-')
19            .map(|num_str| num_str.parse::<u64>().unwrap());
20        let start = bounds.next().unwrap();
21        let end = bounds.next().unwrap();
22        start..=end
23    })
24}

26fn generate_mask(pattern_length: usize, repeats: usize) -> u64 {
27    (0..repeats).fold(0u64, |acc, i| acc + 10u64.pow((i * pattern_length) as u32))
28}

30#[derive(Debug, Clone, Copy, PartialEq, Eq)]
31struct RepeatedPatternInteger {
32    pub value: u64,
33    pub pattern: u64,
34    pub repeats: usize,
35}
36
37impl RepeatedPatternInteger {
38    pub fn try_from_value_and_repeats(value: u64, repeats: usize) -> Result<Self, anyhow::Error> {
39        let digit_count = (value as f32).log10().floor() as usize + 1;
40        if !digit_count.is_multiple_of(repeats) {
41            return Err(anyhow::anyhow!(
42                "Value {} does not have a divisible digit count for pattern length {}",
43                value,
44                repeats
45            ));
46        }
47
48        let pattern_length = digit_count / repeats;
49
50        // Special thanks to Mr Kushagra Raina for suggesting the use of a mask.
51        let mask = generate_mask(pattern_length, repeats);
52
53        if !value.is_multiple_of(mask) {
54            return Err(anyhow::anyhow!(
55                "Value {} is not a repeated pattern integer for repeats {}",
56                value,
57                repeats
58            ));
59        }
60
61        Ok(Self {
62            value,
63            pattern: value / mask,
64            repeats,
65        })
66    }
67
68    pub fn try_from_value(value: u64) -> Result<Self, anyhow::Error> {
69        let digit_count = (value as f32).log10().floor() as usize + 1;
70
71        PRIMES
72            .iter()
73            .filter(|&&r| r <= digit_count)
74            .find_map(|&r| Self::try_from_value_and_repeats(value, r).ok())
75            .ok_or_else(|| anyhow::anyhow!("Value {} is not a repeated pattern integer", value))
76    }
77}
78
79impl TryFrom<u64> for RepeatedPatternInteger {
80    type Error = anyhow::Error;
81
82    fn try_from(value: u64) -> Result<Self, Self::Error> {
83        Self::try_from_value(value)
84    }
85}

87struct RepeatedPatternIntegerCounter {
88    #[cfg(not(feature = "sum-only"))]
89    pub found: Vec<RepeatedPatternInteger>,
90    #[cfg(feature = "sum-only")]
91    pub sum: u64,
92}
93
94impl RepeatedPatternIntegerCounter {
95    #[cfg(not(feature = "sum-only"))]
96    pub fn new() -> Self {
97        Self { found: vec![] }
98    }
99    #[cfg(feature = "sum-only")]
100    pub fn new() -> Self {
101        Self { sum: 0 }
102    }
103
104    pub fn search_iterable_and_add(&mut self, iterable: impl Iterator<Item = u64>) {
105        for item in iterable {
106            // Currently only supports R=2
107            RepeatedPatternInteger::try_from(item)
108                .map(|rpi| {
109                    #[cfg(feature = "sum-only")]
110                    {
111                        self.sum += rpi.value;
112                    }
113                    #[cfg(not(feature = "sum-only"))]
114                    {
115                        self.found.push(rpi);
116                    }
117                })
118                .unwrap_or_default();
119        }
120    }
121
122    pub fn sum(&self) -> u64 {
123        #[cfg(feature = "sum-only")]
124        {
125            self.sum
126        }
127
128        #[cfg(not(feature = "sum-only"))]
129        {
130            self.found.iter().map(|rpi| rpi.value).sum()
131        }
132    }
133}

135fn main() {
136    #[cfg(feature = "jemalloc")]
137    {
138        eprintln!("Using jemalloc as the global allocator");
139    }
140    #[cfg(not(feature = "jemalloc"))]
141    {
142        eprintln!("Using the default global allocator");
143    }
144
145    #[cfg(feature = "profile")]
146    let start_time = Instant::now();
147
148    let iterables = split_input_into_iterables(INPUT);
149
150    let mut counter = RepeatedPatternIntegerCounter::new();
151    for iterable in iterables {
152        #[cfg(feature = "profile-per-loop")]
153        let iteration_time = Instant::now();
154        counter.search_iterable_and_add(iterable);
155        #[cfg(feature = "profile-per-loop")]
156        {
157            eprintln!("Time taken for iteration: {:?}", iteration_time.elapsed());
158        }
159    }
160
161    let sum = counter.sum();
162    println!("Sum of all repeated pattern integers: {}", sum);
163
164    #[cfg(feature = "profile")]
165    {
166        eprintln!("Total time taken: {:?}", start_time.elapsed());
167    }
168}
