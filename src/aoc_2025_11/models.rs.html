15impl DeviceIdToStr for DeviceId {
16    fn to_str(&self) -> String {
17        let c1 = ((*self >> 16) & 0xFF) as u8 as char;
18        let c2 = ((*self >> 8) & 0xFF) as u8 as char;
19        let c3 = (*self & 0xFF) as u8 as char;
20        format!("{}{}{}", c1, c2, c3)
21    }
22}

50impl<'s> traits::IsNode<'s, DeviceId, u32> for Device {
51    fn id(&self) -> &DeviceId {
52        &self.id
53    }
54
55    fn neighbours(
56        &'s self,
57        get_node_by_key: impl Fn(&DeviceId) -> Option<&'s Self>,
58    ) -> impl Iterator<Item = (&'s Self, u32)> {
59        self.connected_devices
60            .iter()
61            .filter_map(move |neighbour_id| {
62                // Currently hardcoding distance as 1 between connected devices,
63                // don't know what Part 2 entails yet
64                // Ignore nodes that no longer exists
65                get_node_by_key(neighbour_id).map(|node| (node, 1))
66            })
67    }
68}

70impl<'s> traits::IsNodeWithIndexedNeighbours<'s, DeviceId, u32> for Device {
71    fn get_neighbour(
72        &'s self,
73        index: usize,
74        get_node_by_key: impl Fn(&DeviceId) -> Option<&'s Self>,
75    ) -> Option<(&'s Self, u32)> {
76        self.connected_devices.get(index).and_then(|neighbour_id| {
77            get_node_by_key(neighbour_id).map(|node| (node, 1))
78        })
79    }

82pub fn invert_device_map(map: &DeviceMap) -> DeviceMap {
83    let mut inverted: DeviceMap = DeviceMap::default();
84
85    for (device_id, device) in map.iter() {
86        inverted
87                .entry(*device_id)
88                .or_insert_with(|| Device::new_empty(*device_id));
89        for neighbour_id in device.connected_devices.iter() {
90            inverted
91                .entry(*neighbour_id)
92                .or_insert_with(|| Device::new_empty(*neighbour_id))
93                .connected_devices
94                .push(*device_id);
95        }
96    }
97
98    inverted
99}

101#[cfg(test)]
102mod test_invert_device_map {
103    use super::*;
104    use crate::parse;
105
106    const INPUT: &str = "svr: aaa bbb
107                         aaa: fft
108                         fft: ccc
109                         bbb: tty
110                         tty: ccc
111                         ccc: ddd eee
112                         ddd: hub
113                         hub: fff
114                         eee: dac
115                         dac: fff
116                         fff: ggg hhh
117                         ggg: out
118                         hhh: out";
119
120    #[test]
121    fn test_invert_device_map() {
122        let devices = parse::text_to_devices(INPUT).expect("Failed to parse devices from input");
123        let inverted = invert_device_map(&devices);
124
125        assert_eq!(inverted[&parse::str_to_device_id("svr")].connected_devices.len(), 0);
126        assert_eq!(inverted[&parse::str_to_device_id("aaa")].connected_devices, vec![parse::str_to_device_id("svr")]);
127        assert_eq!(inverted[&parse::str_to_device_id("bbb")].connected_devices, vec![parse::str_to_device_id("svr")]);
128        assert_eq!(inverted[&parse::str_to_device_id("fft")].connected_devices, vec![parse::str_to_device_id("aaa")]);
129        assert_eq!(inverted[&parse::str_to_device_id("tty")].connected_devices, vec![parse::str_to_device_id("bbb")]);
130        assert_eq!(inverted[&parse::str_to_device_id("ccc")].connected_devices, vec![parse::str_to_device_id("fft"), parse::str_to_device_id("tty")]);
131        assert_eq!(inverted[&parse::str_to_device_id("ddd")].connected_devices, vec![parse::str_to_device_id("ccc")]);
132        assert_eq!(inverted[&parse::str_to_device_id("eee")].connected_devices, vec![parse::str_to_device_id("ccc")]);
133        assert_eq!(inverted[&parse::str_to_device_id("hub")].connected_devices, vec![parse::str_to_device_id("ddd")]);
134        assert_eq!(inverted[&parse::str_to_device_id("dac")].connected_devices, vec![parse::str_to_device_id("eee")]);
135        assert_eq!(inverted[&parse::str_to_device_id("fff")].connected_devices, vec![parse::str_to_device_id("hub"), parse::str_to_device_id("dac")]);
136        assert_eq!(inverted[&parse::str_to_device_id("ggg")].connected_devices, vec![parse::str_to_device_id("fff")]);
137        assert_eq!(inverted[&parse::str_to_device_id("hhh")].connected_devices, vec![parse::str_to_device_id("fff")]);
138        assert_eq!(inverted[&parse::str_to_device_id("out")].connected_devices, vec![parse::str_to_device_id("ggg"), parse::str_to_device_id("hhh")]);
139    }
