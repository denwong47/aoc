5pub struct Orchestrator<'s, B: AddToBuffer> {
6    numeric_buffers: Vec<BufferedLineReader<'s, B>>,
7    operator_buffer: BufferedLineReader<'s, Operator>,
8}

10impl<'s, B: AddToBuffer> Orchestrator<'s, B> {
11    /// Create an Orchestrator from an iterator over lines of text.
12    pub fn from_lines(lines: impl Iterator<Item = &'s str>) -> anyhow::Result<Self> {
13        let mut numeric_lines = Vec::new();
14        let mut operator_lines = Vec::new();
15
16        lines.for_each(|line| match line.chars().next() {
17            Some('0'..='9') => numeric_lines.push(line),
18            Some(' ') => numeric_lines.push(line),
19            Some('+') | Some('*') => operator_lines.push(line),
20            _ => {}
21        });
22
23        if operator_lines.len() != 1 {
24            anyhow::bail!(
25                "Expected exactly one operator line, found {}",
26                operator_lines.len()
27            );
28        }
29
30        let numeric_buffers = numeric_lines
31            .into_iter()
32            .map(|line| BufferedLineReader::new(line))
33            .collect::<Vec<_>>();
34        let operator_buffer = BufferedLineReader::new(operator_lines[0]);
35
36        Ok(Self {
37            numeric_buffers,
38            operator_buffer,
39        })
40    }
41
42    /// Create an Orchestrator from a block of text.
43    pub fn from_text(text: &'s str) -> anyhow::Result<Self> {
44        Self::from_lines(text.lines())
45    }
46}

48impl<'s> Orchestrator<'s, u16> {
49    /// Parse each segment horizonally as a number, then operate on them vertically.
50    pub fn horizontal_process(mut self) -> anyhow::Result<u128> {
51        // We can't use `try_fold` because we need to &mut operator_buffer twice.
52        let mut acc = 0_u128;
53        loop {
54            let operator_char = self.operator_buffer.advance()?;
55            let all_digits: Vec<Option<char>> = {
56                self.numeric_buffers
57                    .iter_mut()
58                    .map(|buf| buf.advance())
59                    .collect::<anyhow::Result<Vec<Option<char>>>>()
60            }?;
61
62            let is_exhausted = operator_char.is_none() && all_digits.iter().all(|c| c.is_none());
63            if is_exhausted
64                || operator_char.is_some_and(|c| c.is_whitespace())
65                    && all_digits
66                        .iter()
67                        .all(|oc| oc.is_some_and(|c| c.is_whitespace()))
68            {
69                #[cfg(feature = "trace")]
70                eprintln!(
71                    "Processing segment with all whitespaces, currently accumulated: {}",
72                    acc
73                );
74
75                // If everything yielded a whitespace, then we know that we have got the columns we
76                // needed. Let's start processing.
77                let numbers = self
78                    .numeric_buffers
79                    .iter_mut()
80                    .map(|buf| buf.yield_buffer());
81
82                let operator = self.operator_buffer.yield_buffer();
83
84                acc = acc
85                    .checked_add(operator.operate_on(numbers)?)
86                    .ok_or_else(|| anyhow::anyhow!("Overflow occurred during accumulation"))?;
87
88                if is_exhausted {
89                    break;
90                }
91            }
92        }
93
94        Ok(acc)
95    }
96}

98impl<'s> Orchestrator<'s, Vec<Option<u8>>> {
99    /// Parse each segment horizonally as a number, then operate on them vertically.
100    pub fn vertical_process(mut self) -> anyhow::Result<u128> {
101        // We can't use `try_fold` because we need to &mut operator_buffer twice.
102        let mut acc = 0_u128;
103        loop {
104            let operator_char = self.operator_buffer.advance()?;
105            let all_digits: Vec<Option<char>> = {
106                self.numeric_buffers
107                    .iter_mut()
108                    .map(|buf| buf.advance())
109                    .collect::<anyhow::Result<Vec<Option<char>>>>()
110            }?;
111
112            let is_exhausted = operator_char.is_none() && all_digits.iter().all(|c| c.is_none());
113            if is_exhausted
114                || operator_char.is_some_and(|c| c.is_whitespace())
115                    && all_digits
116                        .iter()
117                        .all(|oc| oc.is_some_and(|c| c.is_whitespace()))
118            {
119                #[cfg(feature = "trace")]
120                eprintln!(
121                    "Processing segment with all whitespaces, currently accumulated: {}",
122                    acc
123                );
124
125                let numbers = self
126                    .numeric_buffers
127                    .iter_mut()
128                    .try_fold(None, |mut vec, buf| {
129                        let mut digits = buf.yield_buffer();
130
131                        // If the line is not exhausted, we must have inserted a trailing None for teh
132                        // separator (i.e. the whitespace we were checking for above). We need to pop it
133                        // off to avoid messing up multiplication.
134                        if !is_exhausted {
135                            digits.pop_if(|d| d.is_none());
136                        }
137
138                        // Lazy init because we don't know how many numbers there are yet.
139                        if vec.is_none() {
140                            vec = Some(vec![0_u16; digits.len()]);
141                        }
142
143                        digits.iter().enumerate().try_for_each(|(idx, digit)| {
144                            if digit.is_none() {
145                                return Ok(());
146                            }
147
148                            let existing_number = vec.as_ref().expect("Unreachable")[idx];
149                            vec.as_mut().expect("Unreachable")[idx] = existing_number
150                                .checked_mul(10)
151                                .and_then(|v| v.checked_add(digit.unwrap() as u16))
152                                .ok_or_else(|| {
153                                    anyhow!(
154                                        "Overflow when shifting number during vertical processing"
155                                    )
156                                })?;
157
158                            Ok::<_, anyhow::Error>(())
159                        })?;
160
161                        Ok::<_, anyhow::Error>(vec)
162                    })?
163                    .expect("Unreachable: vec should be initialized");
164
165                let operator = self.operator_buffer.yield_buffer();
166
167                acc = acc
168                    .checked_add(operator.operate_on(numbers.into_iter())?)
169                    .ok_or_else(|| anyhow::anyhow!("Overflow occurred during accumulation"))?;
170
171                if is_exhausted {
172                    break;
173                }
174            }
175        }
176
177        Ok(acc)
178    }
179}
